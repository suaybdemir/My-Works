        Scanner input = new Scanner(System.in);
        int a;
        int b = 2;
        int c = 2;
        int d = 0;
                
        do 
        {
            System.out.println(" Lutfen pozitif bir tamsayı giriniz ");
            a = input.nextInt();
        }while(a<1);
        
        while(b<a)
        {
            if(a%b==0)
            {
                d++;
            }
            b++;
        }
        while(c<a+2)
        {
            if(a+2%c==0)
            {
                d++;
            }
            c++;
        }
        if(d==0)
            
        {
            System.out.println(" Bu sayı bir chen asalıdır"+d);
            
        }
        else
            System.out.println(" Bu sayı bir chen asalı degildir ");
